Full Edit (Agent)

POST /api/agent/edit
{
  "example_video": "/path/to/example.mp4",
  "raw_footage": "/path/to/raw.mp4",
  "instructions": "Make it energetic",
  "output_format": "reel",
  "brand": "default",
  "wait": true
}

curl -X POST http://127.0.0.1:12795/api/agent/edit \

  -d '{
    "example_video": "example.mp4",
    "raw_footage": "raw.mp4",
    "output_format": "reel",
    "wait": true
  }'
